"Toggle relative numbers.
set number relativenumber

let s:relative =1
function! ToggleRelative()
    if s:relative  == 0
        let s:relative = 1
        set nonumber
        set number relativenumber

    else
        let s:relative = 0
        set nonumber
        set number norelativenumber

    endif
endfunction
noremap <silent> <leader>n :call ToggleRelative()<CR>
